\title{Folded SBP Operators for Spherical Symmetry:\\
Exact Origin Regularization, SAT Wave Dynamics, and Stability Diagnostics}

\author{SphericalSBPOperators Development Team}
\affiliation{Computational Science and Engineering}

\date{\today}

\begin{abstract}
This report documents the implemented methodology in
\texttt{SphericalSBPOperators.jl} for constructing radial summation-by-parts (SBP)
operators on $[0,R]$ from mirrored Cartesian operators on $[-R,R]$. The method
enforces parity by construction, uses a metric-weighted diagonal mass, and defines
discrete divergence through the SBP identity. A key implementation feature is an exact
near-origin repair of the folded even-gradient operator in rational arithmetic, coupled
to divergence constraints, followed by exact/near-exact verification. The resulting
operators are used in a first-order radial wave system with characteristic SAT boundary
conditions (absorbing, reflecting, Dirichlet). We present the full step-by-step
construction, imposed constraints, and rationale, then derive semidiscrete energy-rate
identities and stability conditions for each boundary closure. Representative numerical
results from the current code path show machine-level SBP residuals, closure-localized
polynomial errors, spectral stability (no exponentially growing modes), and near-zero
energy drift for reflecting SAT with implicit midpoint time integration.
\end{abstract}

\maketitle

\section{Problem Statement and Design Requirements}
We target radial operators on $r\in[0,R]$ for
\begin{equation}
\Divp(u) = \frac{1}{r^p}\frac{d}{dr}\left(r^p u\right), \qquad p\in\{0,1,2\},
\end{equation}
with $p=2$ (spherical), $p=1$ (cylindrical), and $p=0$ (Cartesian radial reduction).

The implementation requirements are:
\begin{enumerate}
  \item one half-grid degree-of-freedom layout for both even/scalar and odd/flux fields;
  \item parity consistency at the origin enforced by extension maps, not ad hoc row edits;
  \item metric-consistent discrete SBP structure
  \begin{equation}
    \Hs D + \Ge^{T}\Hs = B;
  \end{equation}
  \item regular origin treatment consistent with
  $\lim_{r\to 0}\Divp(u)=(p+1)u'(0)$ for odd regular $u$;
  \item practical stability support for wave propagation with characteristic SAT
  boundaries.
\end{enumerate}

\section{Continuum Model and Parity Conditions}
\subsection{Parity}
At $r=0$, parity is a symmetry requirement:
\begin{align}
\phi(-r) &= \phi(r) \quad \text{(even scalar-like field)},\\
u(-r) &= -u(r) \quad \text{(odd radial flux-like field)},
\end{align}
hence $u(0)=0$.

\subsection{Removable singularity at the origin}
For odd regular $u(r)=a_1r+a_3r^3+\cdots$,
\begin{equation}
\lim_{r\to0}\Divp(u)=(p+1)a_1=(p+1)u'(0).
\end{equation}
This condition is explicitly imposed in the discrete divergence first row.

\subsection{First-order radial wave system}
The implemented semidiscrete wave state is $(\Pi,\Xi)$ with
\begin{equation}
\partial_t\Pi = D\Xi,\qquad
\partial_t\Xi = \Ge\Pi.
\end{equation}
At the outer boundary $r=R$, characteristic variables are
\begin{equation}
w_{\mathrm{in}}=\Pi+\Xi,\qquad w_{\mathrm{out}}=\Pi-\Xi.
\end{equation}
SAT penalties are applied on the RHS (not via state projection) according to the chosen
boundary condition.

\section{Folded SBP Construction on \texorpdfstring{$[0,R]$}{[0,R]}}
\subsection{Step 1: mirrored Cartesian operator}
Let the full grid on $[-R,R]$ be
\begin{equation}
\bm{x}^{\mathrm{full}}=(x_1,\ldots,x_M),\qquad x_1=-R,\quad x_M=R,\quad M=2N+1.
\end{equation}
From \texttt{SummationByPartsOperators.jl}, build a Cartesian first-derivative operator
$G^{\mathrm{full}}$ and full mass matrix $H^{\mathrm{full}}$. Matrix extraction supports:
\begin{enumerate}
  \item direct dense conversion when square matrix conversion exists,
  \item probing columns by applying the operator to basis vectors.
\end{enumerate}

\subsection{Step 2: folding maps and half-grid}
Define the nonnegative half-grid indices
\begin{equation}
\mathcal{I}_{+}=\{j \mid x_j\ge 0\},\qquad \bm{r}=(r_1,\ldots,r_{N_h}),\quad r_1=0,\ r_{N_h}=R.
\end{equation}
Build sparse maps
\begin{align}
\Rmat &\in \RR^{N_h\times M} &&\text{(restriction)},\\
\Ee &\in \RR^{M\times N_h} &&\text{(even extension)},\\
\Eo &\in \RR^{M\times N_h} &&\text{(odd extension)}.
\end{align}
For each full-grid point $x_j$, let $\sigma(j)$ satisfy $r_{\sigma(j)}=|x_j|$:
\begin{align}
(\Ee)_{j,\sigma(j)} &= 1,\\
(\Eo)_{j,\sigma(j)} &= \operatorname{sign}(x_j),\qquad \operatorname{sign}(0)=0.
\end{align}
The odd extension therefore enforces zero at the origin by construction.

\subsection{Step 3: folded derivatives}
\begin{align}
\Ge &= \Rmat G^{\mathrm{full}}\Ee,\\
\Go &= \Rmat G^{\mathrm{full}}\Eo.
\end{align}
$\Ge$ maps even fields to odd derivatives; $\Go$ maps odd fields to even derivatives.

\subsection{Step 4: metric mass and boundary operator}
Even-function folding gives
\begin{equation}
\Hc = \frac{1}{2}\Ee^T H^{\mathrm{full}}\Ee.
\end{equation}
Metric weighting produces
\begin{equation}
\Hs = \Hc\diag(r_1^p,\ldots,r_{N_h}^p).
\end{equation}
The boundary matrix has only the outer contribution:
\begin{equation}
B=\diag(0,\ldots,0,R^p).
\end{equation}

\subsection{Step 5: exact coupled near-origin repair of \texorpdfstring{$\Ge$}{Ge}}
\label{sec:repair}
After direct folding, some near-origin rows may retain nonzero first-column coupling
$(\Ge)_{i1}\neq 0$ ($i\ge2$), which contaminates odd/even decoupling. The code repairs
these rows in exact rational arithmetic (\texttt{Rational\{BigInt\}} path).

Define
\begin{equation}
\mathcal{I}_c=\{i\in\{2,\ldots,N_h\}\mid (\Ge)_{i1}\neq 0\}.
\end{equation}
For rows in $\mathcal{I}_c$, unknown coefficients are solved on a stencil
$\mathcal{S}\subset\{2,\ldots,N_h\}$ under two coupled constraint families:

\paragraph*{(a) Even-gradient exactness constraints}
For $m=0,\ldots,q_b$ (with inferred Cartesian boundary exactness $q_b$),
\begin{equation}
\sum_{j\in\mathcal{S}} g_{ij}r_j^{2m}
=
\frac{d}{dr}\left(r^{2m}\right)\Big|_{r=r_i}
\text{(fixed contribution from non-stencil columns)}.
\end{equation}

\paragraph*{(b) Divergence consistency constraints}
For odd degrees $d\in\{1,3,\ldots,q-p\}$ admitted by inferred interior order $q$,
\begin{equation}
\frac{B_{ii}r_i^{d}-\sum_{j}(\Ge)_{ji}H_{jj}r_j^{d}}{H_{ii}}
=(p+d)r_i^{d-1},\qquad i\in\mathcal{I}_c.
\end{equation}

The linear system is solved exactly (Gauss-Jordan over rationals), then verified
exactly. If no exact solution exists for the requested constraint set, the
implementation increases stencil width and, if still necessary, relaxes highest
odd-degree divergence constraints until consistency is reached. This is a deliberate
robustness strategy, not an accidental fallback.

\subsection{Step 6: divergence construction and origin closure}
For rows $i\ge2$:
\begin{equation}
\Hs D = B-\Ge^T\Hs,\qquad
D_{ij}=\frac{(B-\Ge^T\Hs)_{ij}}{(\Hs)_{ii}}.
\end{equation}
At $r=0$, for $p>0$, $(\Hs)_{11}=0$ and SBP does not constrain row 1. The code imposes
\begin{equation}
D_{1,:}=(p+1)(\Go)_{1,:},
\end{equation}
which enforces $(Du)(0)=(p+1)u'(0)$ for odd $u$.

\subsection{Step 7: canonical-to-physical scaling}
Construction is performed on a canonical uniform grid with $\Delta r=1$ and then scaled
to requested physical $R$. With $r=\rho\hat r$:
\begin{align}
\Ge,\Go,D &\mapsto \rho^{-1}(\Ge,\Go,D),\\
\Hs &\mapsto \rho^{p+1}\Hs,\\
B &\mapsto \rho^p B.
\end{align}

\section{What Conditions Are Imposed and Why}
\begin{enumerate}
  \item \textbf{Origin parity}: odd fields satisfy $\Xi(0)=0$, enforced in state and
  RHS space, to keep evolution inside the odd subspace and avoid origin-mode leakage.
  \item \textbf{Origin divergence regularity}: $D_{1,:}=(p+1)(\Go)_{1,:}$ enforces the
  removable singularity limit.
  \item \textbf{SBP structure}: $D$ is derived from $\Hs D+\Ge^T\Hs=B$ for $i\ge2$,
  ensuring summation-by-parts consistency with the metric mass.
  \item \textbf{Near-origin decoupling repair}: exact coupled constraints remove folded
  first-column contamination while preserving polynomial exactness targets.
  \item \textbf{Characteristic SAT closure}: boundary conditions are imposed weakly on
  RHS to control energy (dissipative for absorbing, conservative cancellation for
  reflecting/Dirichlet).
\end{enumerate}

\section{Verification and Diagnostics Workflow}
\subsection{Closure-aware safe interior}
Boundary closure widths are estimated from row-pattern mismatch (and optional
coefficient mismatch) relative to an interior reference row. Right closure is made
conservative by combining pattern-based width with SBP metadata
(\texttt{coefficients.right\_weights}) when available.

Safe interior indices are then
\begin{equation}
\mathcal{I}_{\mathrm{safe}}=\{1+w_{\mathrm{left}},\ldots,N_h-w_{\mathrm{right}}\},
\end{equation}
with additional conservative offset on the left for divergence checks.

\subsection{Validation quantities}
For monomial tests up to selected degree:
\begin{enumerate}
  \item weighted quadrature exactness using $\Hs$;
  \item gradient errors for even monomials under $\Ge$;
  \item divergence errors for odd monomials under $D$;
  \item SBP residual norms of
  $\Rsbp=\Hs D+\Ge^T\Hs-B$
  (full and excluding origin row).
\end{enumerate}
Reported metrics include global max, safe-interior max, and near-boundary max.

\subsection{Extended diagnostics}
The \texttt{diagnose(...)} path additionally checks:
\begin{enumerate}
  \item grid symmetry/uniformity and folded-grid consistency;
  \item algebraic consistency of $(\Rmat,\Ee,\Eo)$ maps;
  \item direct-vs-reconstructed folded operator differences;
  \item localization of polynomial-error hotspots (origin/mid/boundary);
  \item interpreted conclusions (folding issue, closure dominance, origin-dominant divergence, etc.).
\end{enumerate}

\section{Wave-System Discretization and Boundary Conditions}
\subsection{Semidiscrete system}
The implemented RHS is
\begin{align}
\dot{\Pi} &= D\Xi + \sat_{\Pi},\\
\dot{\Xi} &= \Ge\Pi + \sat_{\Xi},
\end{align}
with optional origin projection $\dot{\Xi}(0)=0$.

\subsection{Characteristic SAT penalties at \texorpdfstring{$r=R$}{r=R}}
Define residual $\rho$ and penalties:
\begin{align}
\dot{\Pi}_N &\leftarrow \dot{\Pi}_N - \sigma_{\Pi}H_{NN}^{-1}\rho,\\
\dot{\Xi}_N &\leftarrow \dot{\Xi}_N - \sigma_{\Xi}H_{NN}^{-1}\rho.
\end{align}
Implemented cases:
\begin{align}
\text{absorbing: }&
\rho=w_{\mathrm{in}},\quad \sigma_{\Pi}=\sigma_{\Xi}=B_{NN}/2,\\
\text{reflecting: }&
\rho=w_{\mathrm{in}}-w_{\mathrm{out}}=2\Xi_N,\quad \sigma_{\Pi}=B_{NN}/2,\ \sigma_{\Xi}=0,\\
\text{Dirichlet: }&
\rho=w_{\mathrm{in}}+w_{\mathrm{out}}=2\Pi_N,\quad \sigma_{\Pi}=0,\ \sigma_{\Xi}=B_{NN}/2.
\end{align}
Aliases \texttt{:radiative} and \texttt{:reflective} map to absorbing/reflecting.

\subsection{Jacobian and integration}
The package provides both matrix-state and vector-state ODE forms, analytic Jacobian,
and sparse Jacobian prototype for implicit midpoint solves.

\section{Stability Analysis}
\subsection{Semidiscrete energy identity}
Define
\begin{equation}
E = \frac{1}{2}\left(\Pi^T\Hs\Pi+\Xi^T\Hs\Xi\right).
\end{equation}
For the interior part (without SAT),
\begin{equation}
\dot E_{\mathrm{int}}
= \Pi^T\Hs D\Xi + \Xi^T\Hs\Ge\Pi
= \Pi^T(B+\Rsbp)\Xi,
\end{equation}
where $\Rsbp=\Hs D+\Ge^T\Hs-B$.
Hence
\begin{equation}
\dot E_{\mathrm{int}}=\underbrace{B_{NN}\Pi_N\Xi_N}_{\text{boundary flux}}
\underbrace{\Pi^T\Rsbp\Xi}_{\text{SBP residual term}}.
\end{equation}

\subsection{Boundary-condition-specific energy behavior}
\paragraph*{Reflecting SAT}
SAT contributes $-B_{NN}\Pi_N\Xi_N$, canceling boundary flux:
\begin{equation}
\dot E_{\mathrm{refl}}=\Pi^T\Rsbp\Xi.
\end{equation}
Thus reflecting energy is conservative up to SBP residual and origin-degeneracy effects.

\paragraph*{Dirichlet SAT}
An analogous cancellation occurs, again giving
\begin{equation}
\dot E_{\mathrm{Dir}}=\Pi^T\Rsbp\Xi.
\end{equation}

\paragraph*{Absorbing SAT}
Using $\rho=w_{\mathrm{in}}=\Pi_N+\Xi_N$ with symmetric penalties:
\begin{equation}
\dot E_{\mathrm{abs}}
=
-\frac{B_{NN}}{4}\left(w_{\mathrm{in}}^2+w_{\mathrm{out}}^2\right)
\Pi^T\Rsbp\Xi,
\end{equation}
which is dissipative apart from the SBP residual term.

\subsection{Spectral checks}
The stability suite (\texttt{scripts/StabilityAnalysis.jl}) computes eigenvalues of the
closed block operator
\begin{equation}
L=
\begin{bmatrix}
0 & D\\
\Ge & 0
\end{bmatrix}
\text{SAT terms},
\end{equation}
flags modes with $\operatorname{Re}(\lambda)>$ tolerance, estimates RK4 timestep limits
from the full spectrum, and inspects near-null modes for checkerboard signatures.

\subsection{Runtime consistency checks}
The implementation also probes the runtime RHS operator by basis application and compares
it to the analytic Jacobian operator. For reflecting runs it evaluates:
\begin{enumerate}
  \item direct RHS energy rate $\dot E$,
  \item flux and SBP residual decomposition,
  \item skew defect $S=A^T H_{\mathrm{blk}}+H_{\mathrm{blk}}A$ (full/no-origin/safe blocks),
  \item quadratic-form consistency $\dot E \approx \frac12 U^T S U$.
\end{enumerate}

\subsection{Reflecting SAT drift diagnosis}
For a representative reflecting test (\texttt{R=8}, \texttt{dr=0.2},
\texttt{ImplicitMidpoint}, short horizon):
\begin{itemize}
  \item runtime-vs-analytic operator mismatch: exactly zero in reported max norm;
  \item no-origin skew defect: $1.42\times10^{-14}$;
  \item max $|\dot E|$ over early diagnostic states:
  $6.94\times10^{-17}$ (Gaussian) and $6.07\times10^{-18}$ (compact bump).

This supports the code-level conclusion that semidiscrete reflecting drift is
negligible in these settings, and observed long-time drift (when present) is typically
time-integration or data-consistency related.

\section{Algorithm Summary (Implementation Order)}
\begin{enumerate}
  \item Build full-grid Cartesian SBP operators on $[-R,R]$.
  \item Build $\Rmat,\Ee,\Eo$ and half grid $\bm{r}$ with explicit origin check.
  \item Fold to obtain $\Ge,\Go$.
  \item Form $\Hc$, then metric mass $\Hs=\Hc\diag(r^p)$ and outer-boundary $B$.
  \item Identify contaminated near-origin rows of $\Ge$ and repair them by exact
  coupled constraints (Sec.~\ref{sec:repair}).
  \item Construct divergence from SBP for rows $i\ge2$; set row 1 to
  $(p+1)(\Go)_{1,:}$.
  \item Optionally scale canonical operators to requested physical radius.
  \item Validate (quadrature, polynomial derivative/divergence, SBP residuals) with
  closure-aware safe interior.
  \item Evolve wave system with SAT boundary condition and parity-preserving origin RHS
  constraint.
  \item Run spectral and runtime energy diagnostics for stability assurance.
\end{enumerate}

\section{Conclusions}
The implemented method provides a coherent, publication-grade SBP framework for radial
problems:
\begin{enumerate}
  \item parity-compatible folding from mirrored Cartesian grids,
  \item exact near-origin repair for folded even-gradient consistency,
  \item SBP-consistent divergence with explicit removable-singularity origin closure,
  \item characteristic SAT boundary enforcement with clear energy behavior,
  \item integrated spectral and runtime diagnostics that localize instability sources.
\end{enumerate}
The primary practical lesson is that closure-aware interpretation is essential: global
polynomial errors can be boundary-local while interior consistency remains at machine
precision.
